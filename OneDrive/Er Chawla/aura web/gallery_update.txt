// Build the gallery cards
items.forEach((item, idx) => {
  const card = document.createElement('div');
  card.className = 'g-card'; // Matches your existing CSS
  card.style.cursor = 'pointer';
  card.onclick = () => openLB(idx);
  card.innerHTML = `
    <img src="${item.src}" alt="${item.title}" loading="lazy" style="width:100%; border-radius:8px;">
    <div class="g-label" style="padding:10px; font-size:12px; font-weight:700;">

    </div>
  `;
  container.appendChild(card);
});

let current = 0;

function openLB(idx) {
  current = idx;
  const lbImg = document.getElementById('lb-img');
  const lbCap = document.getElementById('lb-cap');
  
  lbImg.src = items[idx].src;
  lbCap.textContent = items[idx].cat + ' Case ' + String(items[idx].num).padStart(2, '0') + ' — ' + items[idx].title;
  document.getElementById('lightbox').classList.add('open');
  document.body.style.overflow = 'hidden';
}

function closeLB() {
  document.getElementById('lightbox').classList.remove('open');
  document.body.style.overflow = 'auto';
}

// Add Keyboard Support
document.addEventListener('keydown', function(e) {
  const lb = document.getElementById('lightbox');
  if (!lb.classList.contains('open')) return;
  if (e.key === 'Escape') closeLB();
});
